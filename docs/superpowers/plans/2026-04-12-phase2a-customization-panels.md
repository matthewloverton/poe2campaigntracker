# PoE2 Campaign Tracker — Phase 2a: Customization Panels

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add gem tracker, gear advisor, vendor regex panel, inline notes, and settings to the companion mode campaign tracker.

**Architecture:** Extends the existing Tauri + React + TypeScript app with a shared customizations store (Zustand + JSON persistence), three new UI panels wired into the CompanionLayout's sidebar/bottom slots, inline notes on guide steps, and a settings modal. Drag-and-drop reordering uses @dnd-kit. All customization data persists to `customizations.json` via the existing Tauri file I/O commands.

**Tech Stack:** React, TypeScript, Zustand, @dnd-kit/core + @dnd-kit/sortable (drag reordering), existing Tauri file I/O backend

**Design Spec:** `docs/superpowers/specs/2026-04-12-poe2-campaign-tracker-design.md`

**Existing code patterns to follow:**
- CSS Modules with theme variables (--bg-primary, --accent-gold, etc.)
- Zustand stores with async load/save via `invoke("read_user_data"/"write_user_data")`
- Components accept `displayMode` prop for future overlay support
- Title case for names using curated KNOWN_NAMES list
- Underscores replaced with spaces via `displayText()`
- Icons from `/icons/*.png`

```typescript
import type { GemData } from "../types";
import rawGemsData from "./raw/gems.json";

type RawGems = Record<string, Record<string, [number, number]>>;

function transformGems(raw: RawGems): GemData[] {
  const gems: GemData[] = [];
  for (const [category, entries] of Object.entries(raw)) {
    for (const [name, [iconId, rarity]] of Object.entries(entries)) {
      gems.push({
        name,
        category: category as GemData["category"],
        iconId,
        rarity,
      });
    }
  }
  return gems.sort((a, b) => a.name.localeCompare(b.name));
}

export const allGems: GemData[] = transformGems(rawGemsData as RawGems);

export function searchGems(query: string, category?: string): GemData[] {
  const q = query.toLowerCase();
  return allGems.filter((g) => {
    if (category && g.category !== category) return false;
    return g.name.includes(q);
  });
}
```

Create `src/store/customizationsStore.ts`:

```typescript
import { create } from "zustand";
import { invoke } from "@tauri-apps/api/core";
import type {
  Customizations,
  GemPlanEntry,
  GearSlotConfig,
  VendorRegexEntry,
  InlineNote,
} from "../types";
import { DEFAULT_CUSTOMIZATIONS } from "../types/customizations";

const FILENAME = "customizations.json";
let saveTimer: ReturnType<typeof setTimeout> | null = null;

interface CustomizationsState extends Customizations {
  loaded: boolean;
  load: () => Promise<void>;
  save: () => Promise<void>;

  // Gem plan actions
  addGem: (gem: GemPlanEntry) => void;
  removeGem: (id: string) => void;
  updateGem: (id: string, updates: Partial<GemPlanEntry>) => void;
  reorderGems: (ids: string[]) => void;

  // Gear slot actions
  updateGearSlot: (id: string, updates: Partial<GearSlotConfig>) => void;
  reorderGearSlots: (ids: string[]) => void;

  // Vendor regex actions
  addRegex: (entry: VendorRegexEntry) => void;
  removeRegex: (id: string) => void;
  updateRegex: (id: string, updates: Partial<VendorRegexEntry>) => void;

  // Inline note actions
  setNote: (pageIndex: number, stepIndex: number, text: string) => void;
  removeNote: (pageIndex: number, stepIndex: number) => void;
  getNote: (pageIndex: number, stepIndex: number) => string | undefined;
}

function debouncedSave(store: CustomizationsState) {
  if (saveTimer) clearTimeout(saveTimer);
  saveTimer = setTimeout(() => store.save(), 500);
}

export const useCustomizationsStore = create<CustomizationsState>((set, get) => ({
  ...DEFAULT_CUSTOMIZATIONS,
  loaded: false,

  load: async () => {
    try {
      const raw = await invoke<string>("read_user_data", { filename: FILENAME });
      if (raw) {
        const saved = JSON.parse(raw) as Partial<Customizations>;
        set({
          gemPlan: saved.gemPlan ?? DEFAULT_CUSTOMIZATIONS.gemPlan,
          gearSlots: saved.gearSlots ?? DEFAULT_CUSTOMIZATIONS.gearSlots,
          vendorRegexes: saved.vendorRegexes ?? DEFAULT_CUSTOMIZATIONS.vendorRegexes,
          inlineNotes: saved.inlineNotes ?? DEFAULT_CUSTOMIZATIONS.inlineNotes,
          loaded: true,
        });
      } else {
        set({ loaded: true });
      }
    } catch {
      set({ loaded: true });
    }
  },

  save: async () => {
    const { gemPlan, gearSlots, vendorRegexes, inlineNotes } = get();
    const data: Customizations = { gemPlan, gearSlots, vendorRegexes, inlineNotes };
    try {
      await invoke("write_user_data", {
        filename: FILENAME,
        data: JSON.stringify(data, null, 2),
      });
    } catch (e) {
      console.error("Failed to save customizations:", e);
    }
  },

  // Gem plan
  addGem: (gem) => {
    set((s) => ({ gemPlan: [...s.gemPlan, gem] }));
    debouncedSave(get());
  },
  removeGem: (id) => {
    set((s) => ({ gemPlan: s.gemPlan.filter((g) => g.id !== id) }));
    debouncedSave(get());
  },
  updateGem: (id, updates) => {
    set((s) => ({
      gemPlan: s.gemPlan.map((g) => (g.id === id ? { ...g, ...updates } : g)),
    }));
    debouncedSave(get());
  },
  reorderGems: (ids) => {
    set((s) => {
      const byId = new Map(s.gemPlan.map((g) => [g.id, g]));
      const reordered = ids
        .map((id) => byId.get(id))
        .filter((g): g is GemPlanEntry => g !== undefined)
        .map((g, i) => ({ ...g, priority: i }));
      return { gemPlan: reordered };
    });
    debouncedSave(get());
  },

  // Gear slots
  updateGearSlot: (id, updates) => {
    set((s) => ({
      gearSlots: s.gearSlots.map((g) => (g.id === id ? { ...g, ...updates } : g)),
    }));
    debouncedSave(get());
  },
  reorderGearSlots: (ids) => {
    set((s) => {
      const byId = new Map(s.gearSlots.map((g) => [g.id, g]));
      const reordered = ids
        .map((id) => byId.get(id))
        .filter((g): g is GearSlotConfig => g !== undefined)
        .map((g, i) => ({ ...g, priority: i }));
      return { gearSlots: reordered };
    });
    debouncedSave(get());
  },

  // Vendor regexes
  addRegex: (entry) => {
    set((s) => ({ vendorRegexes: [...s.vendorRegexes, entry] }));
    debouncedSave(get());
  },
  removeRegex: (id) => {
    set((s) => ({ vendorRegexes: s.vendorRegexes.filter((r) => r.id !== id) }));
    debouncedSave(get());
  },
  updateRegex: (id, updates) => {
    set((s) => ({
      vendorRegexes: s.vendorRegexes.map((r) => (r.id === id ? { ...r, ...updates } : r)),
    }));
    debouncedSave(get());
  },

  // Inline notes
  setNote: (pageIndex, stepIndex, text) => {
    set((s) => {
      const filtered = s.inlineNotes.filter(
        (n) => !(n.pageIndex === pageIndex && n.stepIndex === stepIndex)
      );
      return { inlineNotes: [...filtered, { pageIndex, stepIndex, text }] };
    });
    debouncedSave(get());
  },
  removeNote: (pageIndex, stepIndex) => {
    set((s) => ({
      inlineNotes: s.inlineNotes.filter(
        (n) => !(n.pageIndex === pageIndex && n.stepIndex === stepIndex)
      ),
    }));
    debouncedSave(get());
  },
  getNote: (pageIndex, stepIndex) => {
    return get().inlineNotes.find(
      (n) => n.pageIndex === pageIndex && n.stepIndex === stepIndex
    )?.text;
  },
}));
```

```tsx
import { ReactNode } from "react";
import {
  DndContext,
  closestCenter,
  KeyboardSensor,
  PointerSensor,
  useSensor,
  useSensors,
  DragEndEvent,
} from "@dnd-kit/core";
import {
  SortableContext,
  sortableKeyboardCoordinates,
  verticalListSortingStrategy,
  useSortable,
} from "@dnd-kit/sortable";
import { CSS } from "@dnd-kit/utilities";

interface DragListProps {
  items: { id: string }[];
  onReorder: (ids: string[]) => void;
  renderItem: (item: { id: string }, index: number) => ReactNode;
}

function SortableItem({ id, children }: { id: string; children: ReactNode }) {
  const { attributes, listeners, setNodeRef, transform, transition, isDragging } = useSortable({ id });
  const style = {
    transform: CSS.Transform.toString(transform),
    transition,
    opacity: isDragging ? 0.5 : 1,
    cursor: "grab",
  };
  return (
    <div ref={setNodeRef} style={style} {...attributes} {...listeners}>
      {children}
    </div>
  );
}

export function DragList({ items, onReorder, renderItem }: DragListProps) {
  const sensors = useSensors(
    useSensor(PointerSensor, { activationConstraint: { distance: 5 } }),
    useSensor(KeyboardSensor, { coordinateGetter: sortableKeyboardCoordinates })
  );

  function handleDragEnd(event: DragEndEvent) {
    const { active, over } = event;
    if (over && active.id !== over.id) {
      const oldIndex = items.findIndex((i) => i.id === active.id);
      const newIndex = items.findIndex((i) => i.id === over.id);
      const newItems = [...items];
      const [moved] = newItems.splice(oldIndex, 1);
      newItems.splice(newIndex, 0, moved);
      onReorder(newItems.map((i) => i.id));
    }
  }

  return (
    <DndContext sensors={sensors} collisionDetection={closestCenter} onDragEnd={handleDragEnd}>
      <SortableContext items={items.map((i) => i.id)} strategy={verticalListSortingStrategy}>
        {items.map((item, index) => (
          <SortableItem key={item.id} id={item.id}>
            {renderItem(item, index)}
          </SortableItem>
        ))}
      </SortableContext>
    </DndContext>
  );
}
```

```tsx
import { useState } from "react";
import { useCustomizationsStore } from "../../store/customizationsStore";
import { DragList } from "../DragList/DragList";
import { GemSearch } from "./GemSearch";
import type { GemPlanEntry, GemData } from "../../types";
import styles from "./GemTracker.module.css";

const STATUS_LABELS: Record<GemPlanEntry["status"], string> = {
  not_acquired: "○",
  acquired: "◆",
  leveling: "▲",
  max_level: "★",
};

const STATUS_CYCLE: GemPlanEntry["status"][] = [
  "not_acquired", "acquired", "leveling", "max_level",
];

export function GemTracker() {
  const gemPlan = useCustomizationsStore((s) => s.gemPlan);
  const addGem = useCustomizationsStore((s) => s.addGem);
  const removeGem = useCustomizationsStore((s) => s.removeGem);
  const updateGem = useCustomizationsStore((s) => s.updateGem);
  const reorderGems = useCustomizationsStore((s) => s.reorderGems);
  const [showSearch, setShowSearch] = useState(false);

  function handleAddGem(gem: GemData) {
    const entry: GemPlanEntry = {
      id: crypto.randomUUID(),
      name: gem.name,
      category: gem.category,
      priority: gemPlan.length,
      source: { type: "vendor" },
      status: "not_acquired",
      cuttingOrder: gemPlan.length,
    };
    addGem(entry);
    setShowSearch(false);
  }

  function cycleStatus(gem: GemPlanEntry) {
    const currentIdx = STATUS_CYCLE.indexOf(gem.status);
    const nextStatus = STATUS_CYCLE[(currentIdx + 1) % STATUS_CYCLE.length];
    updateGem(gem.id, { status: nextStatus });
  }

  return (
    <div className={styles.tracker}>
      <DragList
        items={gemPlan}
        onReorder={reorderGems}
        renderItem={(item) => {
          const gem = gemPlan.find((g) => g.id === item.id)!;
          return (
            <div className={styles.gemRow}>
              <button
                className={styles.statusBtn}
                onClick={(e) => { e.stopPropagation(); cycleStatus(gem); }}
                title={gem.status.replace(/_/g, " ")}
              >
                {STATUS_LABELS[gem.status]}
              </button>
              <span className={styles.gemName}>{gem.name}</span>
              <span className={styles.gemCat}>{gem.category[0].toUpperCase()}</span>
              <button
                className={styles.removeBtn}
                onClick={(e) => { e.stopPropagation(); removeGem(gem.id); }}
              >
                ×
              </button>
            </div>
          );
        }}
      />
      <button className={styles.addBtn} onClick={() => setShowSearch(true)}>
        + Add Gem
      </button>
      {showSearch && (
        <GemSearch onSelect={handleAddGem} onClose={() => setShowSearch(false)} />
      )}
    </div>
  );
}
```
