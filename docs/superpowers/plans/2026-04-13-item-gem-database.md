# Item & Gem Database Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a RePoE-powered equipment and gem database with standalone browsers, build plan integration, and campaign-aware features.

**Architecture:** Build-time Node.js script downloads RePoE JSON data and art assets, transforms into lean typed modules stored in `src/data/raw/`. Data modules in `src/data/` provide indexed lookups and search functions. Two new browser components (ItemBrowser, GemBrowser) render as full-panel views swapped in place of the guide. Build plan editors gain searchable autocomplete pickers backed by real data. Existing free-text fields kept for backward compat.

**Tech Stack:** TypeScript, React 19, Zustand, Vite, Vitest, Node.js native fetch (for transform script)

**Design Spec:** `docs/superpowers/specs/2026-04-13-item-gem-database-design.md`

## Task 2: Transform Script

**Files:**
- Create: `scripts/transform-item-data.mjs`

This is a Node.js ESM script using native fetch (Node 22). It downloads four RePoE JSON files, filters and transforms them, downloads art assets, and writes output to `src/data/raw/` and `public/assets/`.

- [ ] **Step 1: Create the transform script**

```javascript
// scripts/transform-item-data.mjs
import { writeFile, mkdir, access } from "node:fs/promises";
import { dirname, basename } from "node:path";

const REPOE_BASE = "https://repoe-fork.github.io/poe2";

const EQUIPMENT_CLASSES = new Set([
  "Claw", "Dagger", "Wand",
  "One Hand Sword", "Two Hand Sword",
  "One Hand Axe", "Two Hand Axe",
  "One Hand Mace", "Two Hand Mace",
  "Sceptre", "Staff", "Warstaff",
  "Spear", "Flail", "Bow", "Crossbow",
  "Focus", "TrapTool",
  "Body Armour", "Helmet", "Gloves", "Boots",
  "Shield", "Buckler",
  "Amulet", "Ring", "Belt", "Quiver",
]);

// ── Fetch helpers ──────────────────────────────────────

async function fetchJson(filename) {
  const url = `${REPOE_BASE}/${filename}`;
  console.log(`  Fetching ${url}...`);
  const res = await fetch(url);
  if (!res.ok) throw new Error(`Failed to fetch ${url}: ${res.status}`);
  return res.json();
}

async function downloadFile(url, destPath) {
  try {
    await access(destPath);
    return; // already exists
  } catch {
    // file doesn't exist, download it
  }
  await mkdir(dirname(destPath), { recursive: true });
  const res = await fetch(url);
  if (!res.ok) return; // skip missing art
  const buffer = Buffer.from(await res.arrayBuffer());
  await writeFile(destPath, buffer);
}

async function downloadBatch(items, concurrency = 10) {
  let i = 0;
  const total = items.length;
  async function worker() {
    while (i < total) {
      const idx = i++;
      const { url, dest } = items[idx];
      await downloadFile(url, dest);
      if (idx % 50 === 0) console.log(`  Art: ${idx}/${total}`);
    }
  }
  await Promise.all(Array.from({ length: concurrency }, worker));
  console.log(`  Art: ${total}/${total} done`);
}

// ── Transform base items ───────────────────────────────

function transformBaseItems(raw) {
  const items = [];
  for (const [key, val] of Object.entries(raw)) {
    if (val.release_state !== "released") continue;
    if (!EQUIPMENT_CLASSES.has(val.item_class)) continue;

    const props = val.properties || {};
    items.push({
      id: key,
      name: val.name,
      itemClass: val.item_class,
      dropLevel: val.drop_level,
      requirements: val.requirements || { level: 0, strength: 0, dexterity: 0, intelligence: 0 },
      properties: {
        ...(props.physical_damage_min != null && { physicalDamageMin: props.physical_damage_min }),
        ...(props.physical_damage_max != null && { physicalDamageMax: props.physical_damage_max }),
        ...(props.attack_time != null && { attackTime: props.attack_time }),
        ...(props.critical_strike_chance != null && { criticalStrikeChance: props.critical_strike_chance }),
        ...(props.range != null && { range: props.range }),
        ...(props.armour && { armour: props.armour }),
        ...(props.evasion && { evasion: props.evasion }),
        ...(props.energy_shield && { energyShield: props.energy_shield }),
      },
      implicits: val.implicits || [],
      tags: val.tags || [],
      iconPath: ddsToWebpPath(val.visual_identity?.dds_file, "items"),
    });
  }
  return items.sort((a, b) => a.name.localeCompare(b.name));
}

// ── Transform mods ─────────────────────────────────────

function transformMods(raw) {
  const mods = [];
  for (const [key, val] of Object.entries(raw)) {
    if (val.domain !== "item") continue;
    if (val.generation_type !== "prefix" && val.generation_type !== "suffix") continue;
    if (!val.text) continue;

    const canSpawnOn = (val.spawn_weights || [])
      .filter((sw) => sw.weight > 0)
      .map((sw) => sw.tag);
    if (canSpawnOn.length === 0) continue;

    mods.push({
      id: key,
      name: val.name,
      text: val.text,
      type: val.type,
      generationType: val.generation_type,
      group: (val.groups || [])[0] || val.type,
      requiredLevel: val.required_level || 0,
      stats: (val.stats || []).map((s) => ({ id: s.id, min: s.min, max: s.max })),
      canSpawnOn,
    });
  }
  return mods.sort((a, b) => a.type.localeCompare(b.type) || a.requiredLevel - b.requiredLevel);
}

// ── Transform uniques ──────────────────────────────────

function transformUniques(rawUniques, rawMods) {
  // Collect unique mod texts keyed by mod group
  const uniqueModTexts = new Map();
  for (const [, val] of Object.entries(rawMods)) {
    if (val.domain !== "item" || val.generation_type !== "unique") continue;
    if (!val.text) continue;
    const group = (val.groups || [])[0] || val.type;
    if (!uniqueModTexts.has(group)) uniqueModTexts.set(group, []);
    uniqueModTexts.get(group).push(val.text);
  }

  const items = [];
  for (const [key, val] of Object.entries(rawUniques)) {
    if (!EQUIPMENT_CLASSES.has(val.item_class)) continue;
    items.push({
      id: key,
      name: val.name,
      itemClass: val.item_class,
      iconPath: ddsToWebpPath(val.visual_identity?.dds_file, "items"),
      mods: [], // unique mod resolution is best-effort; full mapping requires additional data
    });
  }
  return items.sort((a, b) => a.name.localeCompare(b.name));
}

// ── Transform gems ─────────────────────────────────────

function transformGems(raw) {
  const gems = [];
  for (const [key, val] of Object.entries(raw)) {
    if (!val.base_item || val.base_item.release_state !== "released") continue;

    gems.push({
      id: key,
      name: val.base_item.display_name,
      gemType: val.gem_type,
      color: val.color,
      craftingLevel: val.crafting_level ?? 0,
      craftingTypes: val.crafting_types || [],
      tags: val.tags || [],
      ...(val.support_text && { supportText: val.support_text }),
      recommendedSupports: val.recommended_supports || [],
      requirementWeights: val.requirement_weights || { strength: 0, dexterity: 0, intelligence: 0 },
      iconPath: ddsToWebpPath(val.icon_dds_file, "gems"),
    });
  }
  return gems.sort((a, b) => a.name.localeCompare(b.name));
}

// ── Art path helpers ───────────────────────────────────

function ddsToWebpPath(ddsFile, prefix) {
  if (!ddsFile) return "";
  const filename = basename(ddsFile).replace(/\.dds$/i, ".webp");
  return `${prefix}/${filename}`;
}

function ddsToWebpUrl(ddsFile) {
  if (!ddsFile) return null;
  const path = ddsFile.replace(/\.dds$/i, ".webp");
  return `${REPOE_BASE}/${path}`;
}

// ── Art collection ─────────────────────────────────────

function collectArtDownloads(baseItems, gems, uniques) {
  const seen = new Set();
  const downloads = [];

  function add(ddsFile, prefix) {
    if (!ddsFile) return;
    const url = ddsToWebpUrl(ddsFile);
    const filename = basename(ddsFile).replace(/\.dds$/i, ".webp");
    const dest = `public/assets/${prefix}/${filename}`;
    if (url && !seen.has(dest)) {
      seen.add(dest);
      downloads.push({ url, dest });
    }
  }

  for (const item of baseItems) add(item._ddsFile, "items");
  for (const gem of gems) add(gem._ddsFile, "gems");
  for (const unique of uniques) add(unique._ddsFile, "items");
  return downloads;
}

// ── Main ───────────────────────────────────────────────

async function main() {
  const skipArt = process.argv.includes("--skip-art");

  console.log("Downloading RePoE data...");
  const [rawBaseItems, rawMods, rawUniques, rawGems] = await Promise.all([
    fetchJson("base_items.min.json"),
    fetchJson("mods.min.json"),
    fetchJson("uniques.min.json"),
    fetchJson("skill_gems.min.json"),
  ]);

  console.log("Transforming base items...");
  // Keep _ddsFile temporarily for art download collection
  const baseItemsRaw = [];
  for (const [key, val] of Object.entries(rawBaseItems)) {
    if (val.release_state !== "released" || !EQUIPMENT_CLASSES.has(val.item_class)) continue;
    baseItemsRaw.push({ ...val, _key: key, _ddsFile: val.visual_identity?.dds_file });
  }
  const baseItems = transformBaseItems(rawBaseItems);

  console.log("Transforming mods...");
  const mods = transformMods(rawMods);

  console.log("Transforming uniques...");
  const uniquesRaw = [];
  for (const [key, val] of Object.entries(rawUniques)) {
    if (EQUIPMENT_CLASSES.has(val.item_class)) {
      uniquesRaw.push({ ...val, _key: key, _ddsFile: val.visual_identity?.dds_file });
    }
  }
  const uniques = transformUniques(rawUniques, rawMods);

  console.log("Transforming gems...");
  const gemsRaw = [];
  for (const [key, val] of Object.entries(rawGems)) {
    if (val.base_item?.release_state === "released") {
      gemsRaw.push({ ...val, _key: key, _ddsFile: val.icon_dds_file });
    }
  }
  const gems = transformGems(rawGems);

  console.log(`  ${baseItems.length} base items, ${mods.length} mods, ${uniques.length} uniques, ${gems.length} gems`);

  // Write data files
  await mkdir("src/data/raw", { recursive: true });
  await Promise.all([
    writeFile("src/data/raw/base_items.json", JSON.stringify(baseItems)),
    writeFile("src/data/raw/item_mods.json", JSON.stringify(mods)),
    writeFile("src/data/raw/uniques.json", JSON.stringify(uniques)),
    writeFile("src/data/raw/skill_gems.json", JSON.stringify(gems)),
  ]);
  console.log("Data files written.");

  // Download art
  if (!skipArt) {
    console.log("Downloading art assets...");
    const artDownloads = collectArtDownloads(baseItemsRaw, gemsRaw, uniquesRaw);
    console.log(`  ${artDownloads.length} images to download`);
    await downloadBatch(artDownloads, 15);
  } else {
    console.log("Skipping art downloads (--skip-art)");
  }

  console.log("Done!");
}

main().catch((err) => {
  console.error(err);
  process.exit(1);
});
```

// src/data/items.ts
import type { BaseItem } from "../types/itemDatabase";
import rawItems from "./raw/base_items.json";

export const allItems: BaseItem[] = rawItems as BaseItem[];

export const itemsByClass: Map<string, BaseItem[]> = new Map();
for (const item of allItems) {
  const list = itemsByClass.get(item.itemClass) ?? [];
  list.push(item);
  itemsByClass.set(item.itemClass, list);
}
// Sort each class list by dropLevel
for (const list of itemsByClass.values()) {
  list.sort((a, b) => a.dropLevel - b.dropLevel);
}

export const itemById: Map<string, BaseItem> = new Map(
  allItems.map((item) => [item.id, item])
);

export function getItemsByClass(itemClass: string): BaseItem[] {
  return itemsByClass.get(itemClass) ?? [];
}

// src/data/mods.ts
import type { ItemMod, BaseItem } from "../types/itemDatabase";
import rawMods from "./raw/item_mods.json";

export const allMods: ItemMod[] = rawMods as ItemMod[];

export interface ModGroup {
  type: string;
  label: string;
  tiers: ItemMod[];
}

function modsMatchItem(item: BaseItem): (mod: ItemMod) => boolean {
  const itemTags = new Set(item.tags);
  return (mod) => mod.canSpawnOn.some((tag) => itemTags.has(tag));
}

export function getModsForItem(item: BaseItem): { prefixes: ItemMod[]; suffixes: ItemMod[] } {
  const matches = modsMatchItem(item);
  return {
    prefixes: allMods.filter((m) => m.generationType === "prefix" && matches(m)),
    suffixes: allMods.filter((m) => m.generationType === "suffix" && matches(m)),
  };
}

export function getModsForItemAtLevel(
  item: BaseItem,
  ilvl: number
): { prefixes: ItemMod[]; suffixes: ItemMod[] } {
  const matches = modsMatchItem(item);
  return {
    prefixes: allMods.filter(
      (m) => m.generationType === "prefix" && m.requiredLevel <= ilvl && matches(m)
    ),
    suffixes: allMods.filter(
      (m) => m.generationType === "suffix" && m.requiredLevel <= ilvl && matches(m)
    ),
  };
}

export function groupModsByType(mods: ItemMod[]): ModGroup[] {
  const map = new Map<string, ItemMod[]>();
  for (const mod of mods) {
    const list = map.get(mod.type) ?? [];
    list.push(mod);
    map.set(mod.type, list);
  }

  const groups: ModGroup[] = [];
  for (const [type, tiers] of map) {
    tiers.sort((a, b) => a.requiredLevel - b.requiredLevel);
    // Use the text of the first tier as the group label, stripping the value range
    const label = tiers[0].text.replace(/\([\d-]+\)/g, "#").replace(/\+?[\d-]+/g, "#").trim();
    groups.push({ type, label, tiers });
  }
  return groups.sort((a, b) => a.type.localeCompare(b.type));
}

export function searchMods(query: string, generationType?: "prefix" | "suffix"): ItemMod[] {
  const q = query.toLowerCase();
  return allMods.filter((m) => {
    if (generationType && m.generationType !== generationType) return false;
    return m.text.toLowerCase().includes(q) || m.name.toLowerCase().includes(q);
  });
}

// src/data/gems.ts
import type { GemEntry } from "../types/itemDatabase";
import { gemTier } from "../types/itemDatabase";
import rawGems from "./raw/skill_gems.json";

export const allGems: GemEntry[] = rawGems as GemEntry[];

export const gemById: Map<string, GemEntry> = new Map(
  allGems.map((gem) => [gem.id, gem])
);

// Also index by name for quick lookup
const gemByName: Map<string, GemEntry> = new Map(
  allGems.map((gem) => [gem.name.toLowerCase(), gem])
);

export interface GemSearchFilters {
  gemType?: "active" | "support" | "spirit";
  color?: "r" | "g" | "b" | "w";
  craftingType?: string;
  tags?: string[];
}

export function searchGems(query: string, filters?: GemSearchFilters): GemEntry[] {
  const q = query.toLowerCase();
  return allGems.filter((gem) => {
    if (filters?.gemType && gem.gemType !== filters.gemType) return false;
    if (filters?.color && gem.color !== filters.color) return false;
    if (filters?.craftingType && !gem.craftingTypes.includes(filters.craftingType)) return false;
    if (filters?.tags && !filters.tags.every((t) => gem.tags.includes(t))) return false;
    if (!q) return true;
    return gem.name.toLowerCase().includes(q) || gem.tags.some((t) => t.includes(q));
  });
}

export function getRecommendedSupports(gemId: string): GemEntry[] {
  const gem = gemById.get(gemId);
  if (!gem) return [];
  return gem.recommendedSupports
    .map((id) => gemById.get(id))
    .filter((g): g is GemEntry => g != null);
}

export function getGemsByWeaponType(weaponType: string): GemEntry[] {
  return allGems.filter((g) => g.craftingTypes.includes(weaponType));
}

export function getGemsByTier(gems?: GemEntry[]): Map<number, GemEntry[]> {
  const pool = gems ?? allGems;
  const tiers = new Map<number, GemEntry[]>();
  for (const gem of pool) {
    const tier = gemTier(gem.craftingLevel);
    const list = tiers.get(tier) ?? [];
    list.push(gem);
    tiers.set(tier, list);
  }
  return tiers;
}

export function findGemByName(name: string): GemEntry | undefined {
  return gemByName.get(name.toLowerCase());
}

import type { BaseItem } from "../../types/itemDatabase";

interface ItemDetailProps {
  item: BaseItem;

function formatAttackSpeed(attackTime: number): string {
  return (1000 / attackTime).toFixed(2);
}

function formatCritChance(crit: number): string {
  return (crit / 100).toFixed(1) + "%";
}

function formatDefence(label: string, val?: { min: number; max: number }): string | null {
  if (!val) return null;
  return val.min === val.max ? `${val.min} ${label}` : `${val.min}-${val.max} ${label}`;
}

export function ItemDetail({ item }: ItemDetailProps) {
  const p = item.properties;
  const isWeapon = p.physicalDamageMin != null;

  const reqParts: string[] = [];
  if (item.requirements.strength > 0) reqParts.push(`${item.requirements.strength} Str`);
  if (item.requirements.dexterity > 0) reqParts.push(`${item.requirements.dexterity} Dex`);
  if (item.requirements.intelligence > 0) reqParts.push(`${item.requirements.intelligence} Int`);

  const defences = [
    formatDefence("Armour", p.armour),
    formatDefence("Evasion", p.evasion),
    formatDefence("ES", p.energyShield),
  ].filter(Boolean);

  return (
    <div className={styles.detail}>
      <div className={styles.card}>
        <div className={styles.cardHeader}>
          {item.iconPath && (
            <img
              className={styles.icon}
              src={`/assets/${item.iconPath}`}
              alt={item.name}
              loading="lazy"
            />
          )}
          <div className={styles.cardInfo}>
            <div className={styles.name}>{item.name}</div>
            <div className={styles.meta}>{item.itemClass}</div>
            <div className={styles.stats}>
              {isWeapon && (
                <>
                  <span>{p.physicalDamageMin}-{p.physicalDamageMax} Damage</span>
                  {p.attackTime && <span>{formatAttackSpeed(p.attackTime)} APS</span>}
                  {p.criticalStrikeChance && <span>{formatCritChance(p.criticalStrikeChance)} Crit</span>}
                </>
              )}
              {defences.map((d, i) => <span key={i}>{d}</span>)}
            </div>
            <div className={styles.requirements}>
              <span>Level {item.requirements.level}</span>
              {reqParts.length > 0 && <span>{reqParts.join(", ")}</span>}
            </div>
            {item.implicits.length > 0 && (
              <div className={styles.implicits}>
                {item.implicits.map((imp, i) => (
                  <div key={i} className={styles.implicit}>{imp}</div>
                ))}
              </div>
            )}
          </div>
        </div>
      </div>
      <ModTable item={item} />
    </div>
  );
}
